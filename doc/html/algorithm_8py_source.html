91  def get_graph_from_full_adj_mat(full_adjacency_matrix: np.ndarray):
92  """
93  Gets directed graph based on given full adjacency matrix.
94  Due to igraph limitations, representation of partially directed graph is in object of
95  directed graph, but edges are flagged if these are directed or not.
96  :param full_adjacency_matrix: (list of lists) full adjacency matrix
97  :return: Directed graph with edges flagged if these are directed or not.
98  """
99  # safety check
100  if full_adjacency_matrix.ndim != 2:
101  raise RuntimeError("Wrong argument")
102 
103  # prepare graph g, at first without edges
104  g = Graph(directed=True)
105  vertex_count = full_adjacency_matrix.shape[0]
106  edge_count = full_adjacency_matrix.shape[1]
107  g.add_vertices(vertex_count)
108  edges = np.empty((edge_count, 2), dtype=int)
109  directed_flags = np.empty(edge_count, dtype=bool)
110 
111  # iterate through the matrix & add edges to the graph
112  end_flag = 'e'
113  start_flag = 's'
114  # for every edge...
115  for edge_idx in range(edge_count):
116 
117  new_edge = [] # data structure for edge; consists of 2 tuples: (vertex id , end/start flag)
118  # find vertices incident to the edge
119  for vertex_idx in range(vertex_count):
120  edge_flag = full_adjacency_matrix[vertex_idx][edge_idx]
121  if edge_flag == -1:
122  new_edge.append((vertex_idx, end_flag))
123  elif edge_flag == 1:
124  new_edge.append((vertex_idx, start_flag))
125  else:
126  continue
127  # check if new_edge already consists of 2 vertices
128  if len(new_edge) == 2:
129  break
130 
131  # having obtained new edge, decide if it is directed or undirected
132  if new_edge[0][1] == start_flag and new_edge[1][1] == end_flag:
133  edge_to_add = (new_edge[0][0], new_edge[1][0])
134  directed_flags[edge_idx] = True
135  elif new_edge[0][1] == end_flag and new_edge[1][1] == start_flag:
136  edge_to_add = (new_edge[1][0], new_edge[0][0])
137  directed_flags[edge_idx] = True
138  elif new_edge[0][1] == end_flag and new_edge[1][1] == end_flag:
139  edge_to_add = (new_edge[0][0], new_edge[1][0])
140  directed_flags[edge_idx] = False
141  else:
142  raise RuntimeError("Incorrect structure of full_adjacency_matrix!")
143  edges[edge_idx][0] = edge_to_add[0]
144  edges[edge_idx][1] = edge_to_add[1]
145 
146  g.add_edges(list(map(tuple, edges)))
147  g.es['directed'] = directed_flags.tolist()
148  return g

216  def transform_from_partially_directed(graph_G: PartiallyDirectedGraph):
217  """
218  Transformation from graph G to G1 as in step 2 of the documentation.
219  :return: graph G1 based on G.
220  """
221  g = graph_G.graph.copy()
222 
223  # count correction values because undirected edges are causing miscalculations of vertices' degrees
224  correction_for_vs = defaultdict(int) # default value is zero
225  for edge in g.es:
226  edge["transformed"] = False
227  if not edge["directed"]:
228  correction_for_vs[edge.source] += 1
229  correction_for_vs[edge.target] -= 1
230 
231  # greedy transformation of undirected edges
232  for edge in g.es:
233  if not edge["directed"]:
234  source_vertex_id = edge.source
235  target_vertex_id = edge.target
236  edge_weight = edge["weight"]
237 
238  # degree = incoming - outgoing + correction
239  source_vertex_degree = g.degree(source_vertex_id, type="in") - g.degree(source_vertex_id, type="out") \
240  + correction_for_vs[source_vertex_id]
241  if source_vertex_degree < 0:
242  # transform edge into outgoing
243  # in fact - reverse existing edge: delete current edge and add reversed one
244  g.delete_edges([(source_vertex_id, target_vertex_id)])
245  edge = g.add_edge(target_vertex_id, source_vertex_id)
246  edge["weight"] = edge_weight
247  else:
248  # transform edge into incoming
249  # in fact - leave as it is, just flag it differently
250  pass
251  edge["directed"] = True
252  edge["transformed"] = True
253  correction_for_vs[source_vertex_id] += 1
254  correction_for_vs[target_vertex_id] -= 1
255 
256  return g

419  def GraphBalancing(self, gd : GenericGraph, g2: GenericGraph):
420  """
421  Use Hungarian method to find optimal matching in given bipartite graph. In documentation described at 3d)
422  Then get selcted egdes from biGraph (representing minimal paths in g1)
423  and find edge sequences corresponding to them in graph g1.
424  Duplicate selected edges in g1 - documentation point 3e)
425  :param: gd - complete bipartite graph, g2 - graph with penalty edges
426  :return: penalties - number of added penalty edges
427  """
428  # At start graph G1 dont have penalty edges
429  self.graph.es["IsPenalty"] = False
430 
431  # we need to find minimum matching so to do that invert weight values
432  # method don't support negative weight values
433  for e in gd.graph.es:
434  e["weight"] = 10000 - e["weight"]
435 
436  lstW = gd.graph.es["weight"]
437  matching = gd.graph.maximum_bipartite_matching(weights = lstW)
438 
439 
444 
445  for e in gd.graph.es:
446  e["weight"] = 10000 - e["weight"]
447 
448  penalties = 0
449  for e in matching.edges():
450  # in bipartite graph edges source are vertices with positive degree
451  # so we need to swap target with source
452  trg_lb = gd.graph.vs[e.source]["label"]
453  src_lb = gd.graph.vs[e.target]["label"]
454  src_g2 = g2.graph.vs.find(label=src_lb)
455  trg_g2 = g2.graph.vs.find(label=trg_lb)
456  #print("Src_G2: {}, Trg_G2: {}".format(src_g2["label"], trg_g2["label"]))
457  vertex_sequnce_for_path = g2.graph.get_shortest_paths(src_g2.index, trg_g2.index, "weight")
458  #print(vertex_sequnce_for_path)
459 
460  i, ipre = 0, -1
461  for iVi in vertex_sequnce_for_path[0]:
462  if i > 0:
463  ed1 = self.graph.add_edge(ipre, iVi) #add(duplicate) edge in g1
464  ed2 = g2.graph.es.find(_source=ipre, _target=iVi)
465  ed1["weight"] = ed2["weight"]
466  ed1["directed"] = ed2["directed"]
467  ed1["IsPenalty"] = ed2["IsPenalty"]
468  if ed1["IsPenalty"] == True: penalties += 1
469  i = i + 1
470  ipre = iVi
471 
472  return penalties

475  def FindEuler(self, start_label: str):
476  """
477  Finding Euler cycle in graph G1
478  :param: start_label - starting vertex label
479  :return:
480  """
481  adj = self.graph.get_adjlist()
482  total_weight = 0
483  startVer = self.graph.vs.find(label = start_label)
484  if startVer is None:
485  raise RuntimeError("Incorrect start label in finding Euler cycle")
486 
487  # Hierholzers Algorithm from
488  # https://www.geeksforgeeks.org/hierholzers-algorithm-directed-graph/
489  # not supported in igraph library
490 
491  if len(adj) == 0:
492  return # empty graph
493 
494  # Maintain a stack to keep vertices
495  # We can start from any vertex, here we start with 0
496  curr_path = [startVer.index]
497 
498  # list to store final circuit
499  circuit = []
500 
501  while curr_path:
502 
503  curr_v = curr_path[-1]
504 
505  # If there's remaining edge in adjacency list
506  # of the current vertex
507  if adj[curr_v]:
508 
509  # Find and remove the next vertex that is
510  # adjacent to the current vertex
511  next_v = adj[curr_v].pop()
512 
513  # Push the new vertex to the stack
514  curr_path.append(next_v)
515 
516  # back-track to find remaining circuit
517  else:
518  # Remove the current vertex and
519  # put it in the curcuit
520  circuit.append(curr_path.pop())
521 
522  # we've got the circuit, now print it in reverse
523  print("Euler cycle - postman route: ")
524  for i in range(len(circuit) - 1, -1, -1):
525  #print(circuit[i], end = "")
526 
527  if i > 0:
528  ed = self.graph.es.find(_source=circuit[i], _target=circuit[i-1])
529  total_weight += ed["weight"]
530 
531  # result path printing
532  label = self.graph.vs[circuit[i]]["label"]
533  print(label, end = "")
534  if i:
535  print(" -> ", end = "")
536 
537  print("\nTotal weight: {}".format(total_weight))
538  circuit.reverse()
539  return total_weight, [self.graph.vs[vs]["label"] for vs in circuit], circuit

542  def get_postman_tour(self, useAvgPenalty, penalty, starting_vertex_label, useConstW = False, constW = 80):
543  """
544  The top layer of the algorithm for chinese postman problem.
545  :param useAvgPenalty - if true use add_penaltyAvg_edges otherwise use add_penaltyTm_edges method.
546  :param penalty: penalty for choosing incorrect direction
547  :param starting_vertex_label: label for starting vertex.
548  :return: (cost, tour, iPenCnt) - cost of the tour & tour itself & number of penalty egdes in tour
549  """
550  if not self.is_connected():
551  raise RuntimeError("G1 is not connected.")
552 
553  deg_list = [] # for storing vertices degrees(our degree = DegIn - degOut)
554  if not self.have_euler_tour(deg_list):
555  self.graph.vs["deg"] = deg_list
556  # choosing rule for adding penalty edges
557  if not useAvgPenalty:
558  if not useConstW:
559  g2 = self.add_penaltyTm_edges(penalty) # create G2 as mentioned in documentation
560  else:
561  g2 = self.add_penaltyTm_edges(penalty, useConstW, constW) # create G2 as in test case
562  else:
563  g2 = self.add_penaltyAvg_edges(penalty)
564  print("Graph g2 created")
565  gd, iNeg = self.create_complete_bipart(deg_list, g2)
566  print("bipartite graph generated")
567  if gd is not None:
568  iPenCnt = self.GraphBalancing(gd, g2)
569  print("Finding Euler tour...")
570  cost, tour, tour_by_id = self.FindEuler(starting_vertex_label)
571  else:
572  raise RuntimeError("Uncovered logic path")
573  else:
574  print("Finding Euler...")
575  cost, tour, tour_by_id = self.FindEuler(starting_vertex_label)
576 
577  return cost, tour, tour_by_id, iPenCnt
